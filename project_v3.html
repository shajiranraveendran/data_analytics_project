# Sum of squared distances of samples to their closest cluster center
distortions = []

K = range(1,16,1)

# Loop to find the optimal k
for k in K:
    kmeanModel = KMeans(n_clusters=k)
    kmeanModel.fit(XY_df)
    distortions.append(kmeanModel.inertia_)
    
# Elbow plot
